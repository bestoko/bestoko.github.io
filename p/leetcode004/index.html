		// Ci ‰∏∫Á¨¨i‰∏™Êï∞ÁªÑÁöÑÂâ≤,ÊØîÂ¶ÇC1‰∏∫2Êó∂Ë°®Á§∫Á¨¨1‰∏™Êï∞ÁªÑÂè™Êúâ2‰∏™ÂÖÉÁ¥†„ÄÇLMaxi‰∏∫Á¨¨i‰∏™Êï∞ÁªÑÂâ≤ÂêéÁöÑÂ∑¶ÂÖÉÁ¥†„ÄÇRMini‰∏∫Á¨¨i‰∏™Êï∞ÁªÑÂâ≤ÂêéÁöÑÂè≥ÂÖÉÁ¥†„ÄÇ
		int LMax1, LMax2, RMin1, RMin2, c1, c2, lo = 0, hi = 2 * n;  //Êàë‰ª¨ÁõÆÂâçÊòØËôöÊãüÂä†‰∫Ü'#'ÊâÄ‰ª•Êï∞ÁªÑ1ÊòØ2*nÈïøÂ∫¶
	
		while (lo <= hi)   //‰∫åÂàÜ
		{
			c1 = (lo + hi) / 2;  //c1ÊòØ‰∫åÂàÜÁöÑÁªìÊûú
			c2 = m + n - c1;
	
			LMax1 = (c1 == 0) ? INT_MIN : nums1[(c1 - 1) / 2];
			RMin1 = (c1 == 2 * n) ? INT_MAX : nums1[c1 / 2];
			LMax2 = (c2 == 0) ? INT_MIN : nums2[(c2 - 1) / 2];
			RMin2 = (c2 == 2 * m) ? INT_MAX : nums2[c2 / 2];
	
			if (LMax1 > RMin2)
				hi = c1 - 1;
			else if (LMax2 > RMin1)
				lo = c1 + 1;
			else
				break;
		}
		return (max(LMax1, LMax2) + min(RMin1, RMin2)) / 2.0;
	}
};

{

	vector<int> nums2 = { 1,4,7, 9 };
	
	Solution solution;
	double ret = solution.findMedianSortedArrays(nums1, nums2);
	return 0;
}
